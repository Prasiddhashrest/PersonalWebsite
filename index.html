Interactive Python demonstration showcasing data visualization and analysis capabilities

data_visualization.py 
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# Generate sample economic data
years = np.arange(1990, 2024)
gdp_growth = np.random.normal(2.5, 1.2, len(years))
income_inequality = 30 + np.cumsum(np.random.normal(0.1, 2, len(years)))

# Create DataFrame
data = pd.DataFrame({
    'Year': years,
    'GDP_Growth': gdp_growth,
    'Inequality': income_inequality
})

# Visualize Kuznets Curve relationship
plt.figure(figsize=(10, 6))
plt.scatter(data['GDP_Growth'], 
               data['Inequality'], 
               alpha=0.7, c='teal')
plt.xlabel('GDP Growth Rate (%)')
plt.ylabel('Income Inequality (Gini)')
plt.title('Economic Growth vs Income Inequality')
plt.grid(True, alpha=0.3)
plt.show()
